# 实际实施开发计划

> 本文档是 SkillHub-Enterprise 的详细实施计划，包含具体的开发任务、技术方案、时间安排和验收标准，为开发团队提供可直接执行的操作指南。

## 一、实施原则与策略

### 1.1 核心实施原则

| 原则 | 说明 | 应用场景 |
|------|------|---------|
| **渐进式开发** | 先核心后增强，先基础后高级 | 所有功能模块 |
| **向后兼容** | 新增不修改现有接口和表结构 | API 设计、数据库 |
| **可回滚** | 每个阶段可独立回滚不影响主系统 | 部署策略 |
| **小步快跑** | 每 2 周一个 Sprint，持续集成 | 开发节奏 |
| **测试先行** | 核心功能必须先有测试再有实现 | TDD 流程 |

### 1.2 技术栈确认

**后端技术栈：**
- 核心框架：Java 21 + Spring Boot 3.x
- 数据库：PostgreSQL 16 + pgvector 扩展
- 缓存：Redis 7
- 对象存储：MinIO
- AI 集成：LangChain4j 1.x
- 向量检索：pgvector (IVFFlat/HNSW)
  
**前端技术栈：**
- 框架：React 19 + TypeScript
- UI 库：Ant Design Pro 6.x（渐进引入）
- 状态管理：TanStack Query
- 路由：React Router 6
- 构建工具：Vite

**基础设施：**
- 容器化：Docker + Docker Compose
- 监控：OpenTelemetry + Prometheus
- 日志：ELK (Elasticsearch + Logstash + Kibana)

```java
public class SkillDimensionService {

    public List<SkillDimensionTreeVO> getDimensionTree() {
        List<SkillDimension> roots = dimensionRepository
            .findByParentIdIsNullAndStatusOrderBySortOrder("active");
        return roots.stream().map(this::buildTree).collect(Collectors.toList());

    public void mapSkillToDimension(Long skillId, Long dimensionId, String value) {
        Skill skill = skillRepository.findById(skillId)
            .orElseThrow(() -> new ResourceNotFoundException("Skill not found"));
        SkillDimension dimension = dimensionRepository.findById(dimensionId)
            .orElseThrow(() -> new ResourceNotFoundException("Dimension not found"));
        SkillDimensionMapping mapping = new SkillDimensionMapping();
        mapping.setSkill(skill);
        mapping.setDimension(dimension);
        mapping.setValue(value);
        dimensionMappingRepository.save(mapping);
    }

    public void removeSkillDimensionMapping(Long skillId, Long dimensionId) {
        dimensionMappingRepository.deleteBySkillIdAndDimensionId(skillId, dimensionId);
    }

    public Page<Skill> getSkillsByDimension(Long dimensionId, String value, Pageable pageable) {
        // 查询指定维度下的技能
        return dimensionMappingRepository.findByDimensionIdAndValue(dimensionId, value, pageable)
            .map(SkillDimensionMapping::getSkill);
    }
}
```

```java
@Service
public class EmbeddingService {

    @Value("${skillhub.embedding.model:minimax}")
    private String embeddingModel;

    @Value("${skillhub.embedding.minimax.api-key:}")
    private String minimaxApiKey;

    @Value("${skillhub.embedding.minimax.base-url:}")
    private String minimaxBaseUrl;

    @Value("${skillhub.embedding.minimax.model:emb-0}")
    private String minimaxModel;

    @Value("${skillhub.embedding.minimax.dimensions:1024}")
    private int dimensions;

    private EmbeddingModel embeddingModelImpl;

    @PostConstruct
    public void init() {
        if ("minimax".equals(embeddingModel)) {
            // MiniMax 使用 OpenAI 兼容 API
            OpenAiEmbeddingModel model = OpenAiEmbeddingModel.builder()
                .apiKey(minimaxApiKey)
                .baseUrl(minimaxBaseUrl.isEmpty() ? "https://api.minimaxi.com/v1" : minimaxBaseUrl)
                .model(minimaxModel)
                .dimensions(dimensions)
                .build();
            embeddingModelImpl = model;
        } else if ("local".equals(embeddingModel)) {
            // 本地模型支持（如 Ollama）
            embeddingModelImpl = new LocalEmbeddingModel();
        }
    }

    public float[] embed(String text) {
        if (embeddingModelImpl == null) {
            throw new IllegalStateException("Embedding model not initialized");
        }
        EmbeddingResult result = embeddingModelImpl.embed(text);
        return result.contentAsVector();
    }

    public List<float[]> embedBatch(List<String> texts) {
        return texts.stream().map(this::embed).collect(Collectors.toList());
    }
}
```

**VectorSearchService：**

```java
@Service
public class VectorSearchService {

    @Autowired
    private EntityManager entityManager;

    @Autowired
    private EmbeddingService embeddingService;

    /**
     * 为技能生成并存储向量
     */
    @Transactional
    public void indexSkill(Skill skill) {
        String textToEmbed = buildSkillText(skill);
        float[] embedding = embeddingService.embed(textToEmbed);

        entityManager.createNativeQuery(
            "UPDATE skills SET embedding = ? WHERE id = ?"
        ).setParameter(1, FloatArrayType.toPGvector(embedding))
         .setParameter(2, skill.getId())
         .executeUpdate();
    }

    /**
     * 语义搜索技能
     */
    public Page<Skill> semanticSearch(String query, int page, int size) {
        float[] queryEmbedding = embeddingService.embed(query);

        Query nativeQuery = entityManager.createNativeQuery(
            """
            SELECT s.*, 1 - (s.embedding <=> :query) as similarity
            FROM skills s
            WHERE s.embedding IS NOT NULL
            ORDER BY s.embedding <=> :query
            LIMIT :limit OFFSET :offset
            """,
            Skill.class
        );
        nativeQuery.setParameter("query", FloatArrayType.toPGvector(queryEmbedding));
        nativeQuery.setParameter("limit", size);
        nativeQuery.setParameter("offset", page * size);

        @SuppressWarnings("unchecked")
        List<Skill> results = nativeQuery.getResultList();
        return new PageImpl<>(results, PageRequest.of(page, size), results.size());
    }

    /**
     * 混合搜索（向量 + 关键词）
     */
    public Page<Skill> hybridSearch(String query, double vectorWeight, int page, int size) {
        float[] queryEmbedding = embeddingService.embed(query);

        Query nativeQuery = entityManager.createNativeQuery(
            """
            SELECT s.*,
                   (1 - (s.embedding <=> :query)) * :vectorWeight +
                   ts_rank(to_tsvector('simple', s.name || ' ' || COALESCE(s.description, '')), plainto_tsquery('simple', :keyword)) * :keywordWeight as combined_rank
            FROM skills s
            WHERE s.embedding IS NOT NULL
              AND (s.name ILIKE :keyword OR COALESCE(s.description, '') ILIKE :keyword)
            ORDER BY combined_rank DESC
            LIMIT :limit OFFSET :offset
            """,
            Skill.class
        );
        nativeQuery.setParameter("query", FloatArrayType.toPGvector(queryEmbedding));
        nativeQuery.setParameter("keyword", "%" + query + "%");
        nativeQuery.setParameter("vectorWeight", vectorWeight);
        nativeQuery.setParameter("keywordWeight", 1 - vectorWeight);
        nativeQuery.setParameter("limit", size);
        nativeQuery.setParameter("offset", page * size);

        @SuppressWarnings("unchecked")
        List<Skill> results = nativeQuery.getResultList();
        return new PageImpl<>(results, PageRequest.of(page, size), results.size());
    }

    private String buildSkillText(Skill skill) {
        StringBuilder sb = new StringBuilder();
        sb.append(skill.getName()).append(" ");
        if (skill.getDescription() != null) {
            sb.append(skill.getDescription()).append(" ");
        }
        if (skill.getLabels() != null) {
            sb.append(skill.getLabels().stream()
                .map(Label::getName)
                .collect(Collectors.joining(" ")));
        }
        return sb.toString();
    }
}
```

#### 2.4.2 文档解析服务

**DocumentParserService：**

```java
@Service
public class DocumentParserService {

    private static final Set<String> SUPPORTED_EXTENSIONS = Set.of(
        ".pdf", ".docx", ".doc", ".md", ".markdown", ".txt", ".html", ".pptx"
    );

    @Autowired
    private EmbeddingService embeddingService;

    /**
     * 解析文档并分段
     */
    public List<DocumentChunk> parseAndChunk(InputStream inputStream, String filename) {
        String extension = getExtension(filename);
        String content = parseDocument(inputStream, extension);

        // 使用语义分段（基于 LLM）
        return semanticChunk(content, filename);
    }

    private String parseDocument(InputStream inputStream, String extension) {
        return switch (extension.toLowerCase()) {
            case ".pdf" -> parsePdf(inputStream);
            case ".docx" -> parseDocx(inputStream);
            case ".md", ".markdown" -> parseMarkdown(inputStream);
            case ".txt" -> parseTxt(inputStream);
            case ".html" -> parseHtml(inputStream);
            default -> throw new UnsupportedDocumentException("Unsupported format: " + extension);
        };
    }

    /**
     * 语义分段（基于 LLM）
     */
    private List<DocumentChunk> semanticChunk(String content, String filename) {
        // 按段落分割
        List<String> paragraphs = splitIntoParagraphs(content);

        // 合并小段落
        List<String> chunks = new ArrayList<>();
        StringBuilder currentChunk = new StringBuilder();

        for (String paragraph : paragraphs) {
            if (currentChunk.length() + paragraph.length() < 1000) {
                currentChunk.append(paragraph).append("\n\n");
            } else {
                if (currentChunk.length() > 0) {
                    chunks.add(currentChunk.toString().trim());
                }
                currentChunk = new StringBuilder(paragraph).append("\n\n");
            }
        }
        if (currentChunk.length() > 0) {
            chunks.add(currentChunk.toString().trim());
        }

        return chunks.stream()
            .map(chunk -> {
                DocumentChunk dc = new DocumentChunk();
                dc.setContent(chunk);
                dc.setTitle(extractTitle(chunk, filename));
                dc.setContentHash(hashContent(chunk));
                dc.setMetadata(Map.of("filename", filename));
                return dc;
            })
            .collect(Collectors.toList());
    }
}
```

```java
     * 上传文档到知识库

    @Transactional
    public List<KnowledgeChunk> addDocument(Long libraryId, InputStream inputStream,
                                            String filename, Long userId) {
        KnowledgeLibrary library = libraryRepository.findById(libraryId)
            .orElseThrow(() -> new ResourceNotFoundException("Library not found"));

        // 权限检查
        aclService.checkWriteAccess(libraryId, userId);

        // 解析并分段
        List<DocumentChunk> rawChunks = parserService.parseAndChunk(inputStream, filename);

        // 向量化并保存
        List<KnowledgeChunk> savedChunks = new ArrayList<>();
        for (DocumentChunk raw : rawChunks) {
            // 检查是否重复
            if (chunkRepository.existsByLibraryIdAndContentHash(libraryId, raw.getContentHash())) {
                continue; // 跳过重复内容
            }

            KnowledgeChunk chunk = new KnowledgeChunk();
            chunk.setLibraryId(libraryId);
            chunk.setTitle(raw.getTitle());
            chunk.setContent(raw.getContent());
            chunk.setContentHash(raw.getContentHash());
            chunk.setMetadata(raw.getMetadata());
            chunk.setTokenCount(estimateTokenCount(raw.getContent()));

            // 生成向量
            float[] embedding = embeddingService.embed(raw.getContent());
            chunk.setEmbedding(embedding);

            savedChunks.add(chunkRepository.save(chunk));
        }

        // 更新知识库统计
        library.setStatsItems(library.getStatsItems() + savedChunks.size());
        libraryRepository.save(library);

        return savedChunks;
    }

    /**
     * 知识库检索（RAG）
     */
    public List<ChunkSearchResult> searchLibrary(Long libraryId, String query,
                                                  int topK, Long userId) {
        // 权限检查
        aclService.checkReadAccess(libraryId, userId);

        float[] queryEmbedding = embeddingService.embed(query);

        // 向量检索
        List<Object[]> results = entityManager.createNativeQuery(
            """
            SELECT kc.id, kc.title, kc.content,
                   1 - (kc.embedding <=> :query) as similarity
            FROM knowledge_chunks kc
            WHERE kc.library_id = :libraryId
            ORDER BY kc.embedding <=> :query
            LIMIT :topK
            """
        ).setParameter("query", toPGvector(queryEmbedding))
         .setParameter("libraryId", libraryId)
         .setParameter("topK", topK)
         .getResultList();

        return results.stream().map(row -> {
            ChunkSearchResult result = new ChunkSearchResult();
            result.setChunkId(((Number) row[0]).longValue());
            result.setTitle((String) row[1]);
            result.setContent((String) row[2]);
            result.setSimilarity(((Number) row[3]).doubleValue());
            return result;
        }).collect(Collectors.toList());
    }
```

```java
@Service
public class GitAnalysisService {

    @Autowired
    private GitDataSourceRepository dataSourceRepository;

    @Autowired
    private GitCommitRepository commitRepository;

    @Autowired
    private LlmService llmService;

    /**
     * 同步 Git 提交记录
     */
    @Async
    public void syncCommits(Long dataSourceId) {
        GitDataSource dataSource = dataSourceRepository.findById(dataSourceId)
            .orElseThrow(() -> new ResourceNotFoundException("DataSource not found"));

        GitClient client = gitClientFactory.create(dataSource);
        List<CommitInfo> commits = client.fetchCommits(dataSource.getRepositoryUrl(),
                                                        dataSource.getFilterPatterns(),
                                                        dataSource.getExcludePatterns());

        for (CommitInfo commit : commits) {
            // 去重保存
            if (!commitRepository.existsByDataSourceIdAndCommitHash(dataSourceId, commit.getHash())) {
                GitCommit entity = new GitCommit();
                entity.setDataSourceId(dataSourceId);
                entity.setCommitHash(commit.getHash());
                entity.setAuthorName(commit.getAuthorName());
                entity.setAuthorEmail(commit.getAuthorEmail());
                entity.setCommittedAt(commit.getCommittedAt());
                entity.setMessage(commit.getMessage());
                entity.setFilesChanged(commit.getFilesChanged());
                entity.setInsertions(commit.getInsertions());
                entity.setDeletions(commit.getDeletions());
                commitRepository.save(entity);
            }
        }

        dataSource.setLastSyncAt(java.time.LocalDateTime.now());
        dataSourceRepository.save(dataSource);
    }

    /**
     * LLM 分析提交（生成技术摘要）
     */
    @Transactional
    public void analyzeCommit(Long commitId) {
        GitCommit commit = commitRepository.findById(commitId)
            .orElseThrow(() -> new ResourceNotFoundException("Commit not found"));

        if (commit.getProcessed()) {
            return;
        }

        // 获取提交的文件变更详情
        String diffContent = gitClientFactory.create(commit.getDataSource())
            .getCommitDiff(commit.getCommitHash());

        // 调用 LLM 分析
        String prompt = String.format("""
            分析以下代码提交，提取：
            1. 技术的技术摘要（100字内）
            2. 涉及的技术栈/技能标签
            3. 是否有可复用的设计模式或最佳实践

            提交信息：%s
            变更统计：+%d/-%d 文件

            返回 JSON 格式：
            {
              "summary": "...",
              "skills": ["java", "spring", "..."],
              "patterns": ["..."]
            }
            """,
            commit.getMessage(),
            commit.getInsertions(),
            commit.getDeletions());

        LlmAnalysisResult result = llmService.analyze(prompt, diffContent);

        commit.setLlmSummary(result.getSummary());
        commit.setLlmSkills(result.getSkills());
        commit.setProcessed(true);
        commitRepository.save(commit);
    }

    /**
     * 批量处理待分析的提交
     */
    @Scheduled(fixedDelay = 60000)  // 每分钟执行
    public void processPendingCommits() {
        List<GitCommit> pending = commitRepository.findTop100ByProcessedFalse();
        for (GitCommit commit : pending) {
            try {
                analyzeCommit(commit.getId());
            } catch (Exception e) {
                log.error("Failed to analyze commit {}", commit.getId(), e);
            }
        }
    }
}
```

```java
@Service
public class BehaviorAnalysisService {

    @Autowired
    private BehaviorEventRepository eventRepository;

    @Autowired
    private BehaviorPatternRepository patternRepository;

    @Autowired
    private LlmService llmService;

    /**
     * 分析用户行为模式
     */
    @Transactional
    public void analyzeUserPatterns(Long userId) {
        // 收集最近 30 天的行为数据
        LocalDateTime since = LocalDateTime.now().minusDays(30);
        List<BehaviorEvent> events = eventRepository.findByUserIdAndTimestampAfter(userId, since);

        if (events.isEmpty()) {
            return;
        }

        // 按类型聚合统计数据
        Map<String, Long> eventTypeStats = events.stream()
            .collect(Collectors.groupingBy(BehaviorEvent::getEventType, Collectors.counting()));

        Map<String, Long> languageStats = events.stream()
            .filter(e -> e.getLanguage() != null)
            .collect(Collectors.groupingBy(BehaviorEvent::getLanguage, Collectors.counting()));

        // 调用 LLM 进行模式识别
        String prompt = String.format("""
            基于以下用户行为统计数据，分析用户的：
            1. 主要工作模式（如：快速迭代型/稳重设计型/实验探索型）
            2. 技能专长领域（擅长的技术栈）
            3. 学习成长模式

            行为统计：
            - 事件类型分布：%s
            - 编程语言分布：%s
            - 总事件数：%d

            返回 JSON：
            {
              "workingStyle": {"type": "...", "confidence": 85, "evidence": ["..."]},
              "skillStrengths": [{"skill": "java", "level": "expert", "evidence": ["..."]}],
              "learningMode": {"type": "...", "confidence": 75, "evidence": ["..."]}
            }
            """,
            eventTypeStats, languageStats, events.size());

        LlmPatternAnalysis result = llmService.analyzePatterns(prompt);

        // 保存分析结果
        savePattern(userId, "working_style", result.getWorkingStyle());
        savePattern(userId, "skill_strength", result.getSkillStrengths());
        savePattern(userId, "learning_mode", result.getLearningMode());
    }

    /**
     * 基于行为模式推荐技能
     */
    public List<SkillRecommendation> recommendSkills(Long userId) {

            .findTopByUserIdAndPatternTypeOrderByAnalyzedAtDesc(userId, "skill_strength")
            .orElse(null);

        if (pattern == null) {
            return Collections.emptyList();
        }

        // 根据技能专长推荐相关或进阶技能
        List<String> strengths = extractSkills(pattern);
        return skillService.findRelatedSkills(strengths);
    }
```

```java
@Service
public class ContainerPoolService {

    private final Map<String, Container> activeContainers = new ConcurrentHashMap<>();
    private final BlockingQueue<String> availableContainers = new LinkedBlockingQueue<>();

    @Value("${skillhub.executor.container.pool-size:5}")
    private int poolSize;

    @Value("${skillhub.executor.container.image:skillhub/executor:latest}")
    private String containerImage;

    @PostConstruct
    public void initPool() {
        for (int i = 0; i < poolSize; i++) {
            String containerId = createContainer();
            availableContainers.offer(containerId);
        }
    }

    /**
     * 租用容器
     */
    public ContainerLease lease(int timeoutSeconds) {
        String containerId = availableContainers.poll(timeoutSeconds, TimeUnit.SECONDS);
        if (containerId == null) {
            throw new ContainerPoolExhaustedException("No containers available");
        }

        Container container = dockerClient.getContainer(containerId);
        activeContainers.put(containerId, container);

        return new ContainerLease(containerId, container, this);
    }

    /**
     * 归还容器
     */
    public void release(String containerId) {
        activeContainers.remove(containerId);
        resetContainer(containerId);
        availableContainers.offer(containerId);
    }

    /**
     * 执行技能
     */
    public ExecutionResult execute(ContainerLease lease, SkillPackage skillPackage,
                                     Map<String, String> parameters) {
        String containerId = lease.getContainerId();

        try {
            // 上传技能包到容器
            uploadToContainer(containerId, skillPackage);

            // 设置资源限制
            setResourceLimits(containerId, skillPackage.getResourceLimits());

            // 执行技能
            String result = dockerClient.exec(containerId,
                "/opt/executor/run-skill.sh " + skillPackage.getEntryPoint(),
                parameters);

            return ExecutionResult.success(result);
        } catch (Exception e) {
            return ExecutionResult.failure(e.getMessage());
        }
    }
```

```sql
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE git_commits (
    id BIGSERIAL PRIMARY KEY,
    data_source_id BIGINT NOT NULL REFERENCES git_data_sources(id) ON DELETE CASCADE,
    commit_hash VARCHAR(40) NOT NULL,
    author_name VARCHAR(100),
    author_email VARCHAR(100),
    committed_at TIMESTAMP,
    message TEXT,
    files_changed INTEGER,
    insertions INTEGER,
    deletions INTEGER,
    llm_summary TEXT,
    llm_skills TEXT[],
    processed BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(data_source_id, commit_hash)
);

CREATE TABLE behavior_events (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    event_type VARCHAR(50) NOT NULL,
    duration_ms BIGINT,
    file_path VARCHAR(500),
    language VARCHAR(50),
    project_name VARCHAR(100),
    timestamp TIMESTAMP NOT NULL,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE behavior_patterns (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(id),
    pattern_type VARCHAR(50) NOT NULL,
    pattern_name VARCHAR(100),
    description TEXT,
    confidence DECIMAL(5,2),
    evidence JSONB,
    analyzed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
